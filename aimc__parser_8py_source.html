56def parse_mapping_configurations(aimc_submodel: model.Submodel) -> MappingConfigurations | None:
57 """Parse all mapping configurations in the AIMC submodel.
58
59 :param aimc_submodel: The AIMC submodel to parse mapping configurations from.
60 :return: A list of parsed mapping configurations or None if parsing failed.
61 """
62 logger.info("Parse mapping configurations from AIMC submodel.")
63
64 mapping_configurations: list[MappingConfiguration] = []
65
66 # get all mapping configuration elements
67 mapping_configurations_elements = get_mapping_configuration_elements(aimc_submodel)
68 if mapping_configurations_elements is None:
69 logger.error("No mapping configuration elements found in AIMC submodel.")
70 return None
71
72 # parse each mapping configuration element
73 for mc_element in mapping_configurations_elements:
74 mc = parse_mapping_configuration_element(mc_element)
75 if mc is not None:
76 mapping_configurations.append(mc)
77
78 logger.debug(f"Parsed {len(mapping_configurations)} mapping configurations.")
79
80 mcs = MappingConfigurations()
81 mcs.configurations = mapping_configurations
82 # add all unique AID submodel IDs from all mapping configurations
83 mcs.aid_submodel_ids = list({mc.aid_submodel_id for mc in mapping_configurations})
84
85 logger.debug(f"Found {len(mcs.aid_submodel_ids)} unique AID submodel IDs in mapping configurations.")
86 logger.debug(f"Found {len(mcs.configurations)} mapping configurations in AIMC submodel.")
87
88 return mcs

158def _generate_source_sink_relations(mapping_configuration_element: model.SubmodelElementCollection) -> list[SourceSinkRelation]:
159 source_sink_relations: list[SourceSinkRelation] = []
160
161 logger.debug(f"Get 'MappingSourceSinkRelations' from mapping configuration '{mapping_configuration_element}'.")
162
163 relations_list: model.SubmodelElementList = ch.find_by_in_semantic_id(
164 mapping_configuration_element, "/idta/AssetInterfacesMappingConfiguration/1/0/MappingSourceSinkRelations"
165 )
166
167 if relations_list is None or not isinstance(relations_list, model.SubmodelElementList):
168 logger.error(f"'MappingSourceSinkRelations' not found in mapping configuration '{mapping_configuration_element.id_short}'.")
169 return source_sink_relations
170
171 for source_sink_relation in relations_list.value:
172 logger.debug(f"Parse source-sink relation '{source_sink_relation}'.")
173
174 if not isinstance(source_sink_relation, model.RelationshipElement):
175 logger.warning(f"'{source_sink_relation}' is not a RelationshipElement")
176 continue
177
178 if source_sink_relation.first is None or len(source_sink_relation.first.key) == 0:
179 logger.warning(f"'first' reference is missing in RelationshipElement '{source_sink_relation.id_short}'")
180 continue
181
182 if source_sink_relation.second is None or len(source_sink_relation.second.key) == 0:
183 logger.warning(f"'second' reference is missing in RelationshipElement '{source_sink_relation.id_short}'")
184 continue
185
186 relation = SourceSinkRelation()
187 relation.source_properties = _get_reference_properties(source_sink_relation.first)
188 relation.sink_properties = _get_reference_properties(source_sink_relation.second)
189
190 source_sink_relations.append(relation)
191
192 return source_sink_relations
