83 def parse_properties(self, aid_interface: model.SubmodelElementCollection) -> dict[str, PropertyDetails]:
84 """Find all first-level and nested properties in a provided SMC describing one interface in the AID.
85
86 Map each property (either top-level or nested) to the according 'href' attribute.
87 Nested properties are further mapped to the hierarchical list of keys
88 that are necessary to extract their value from the payload of the interface.
89
90 :param aid_interface: An SMC describing an interface in the AID.
91 :return: A dictionary mapping each property (represented by its idShort-path) to PropertyDetails.
92 """
93 mapping: dict[str, PropertyDetails] = {}
94
95 interaction_metadata: model.SubmodelElementCollection | None = find_by_semantic_id(
96 aid_interface.value, "https://admin-shell.io/idta/AssetInterfacesDescription/1/0/InteractionMetadata"
97 )
98 if interaction_metadata is None:
99 raise ValueError(f"'InteractionMetadata' SMC not found in the provided '{aid_interface.id_short}' SMC.")
100
101 properties: model.SubmodelElementCollection | None = find_by_semantic_id(
102 interaction_metadata.value, "https://www.w3.org/2019/wot/td#PropertyAffordance"
103 )
104 if properties is None:
105 raise ValueError("'properties' SMC not found in 'InteractionMetadata' SMC.")
106
107 fl_properties: list[model.SubmodelElement] = find_all_by_semantic_id(
108 properties.value, "https://admin-shell.io/idta/AssetInterfacesDescription/1/0/PropertyDefinition"
109 )
110 if fl_properties is None:
111 print("WARN: No first-level 'property' SMC not found in 'properties' SMC.")
112 return {}

114 def traverse_property(
115 smc: model.SubmodelElementCollection,
116 parent_path: str,
117 href: str,
118 key_path: list[str | int],
119 is_items=False,
120 idx=None,
121 is_top_level=False,
122 protocol_binding: IProtocolBinding = None,
123 ):
124 # determine local key only if not top-level
125 if not is_top_level:
126 if is_items and idx is not None:
127 # is a nested "items" property -> add index to the list of keys
128 local_key = idx
129 else:
130 # is a nested "properties" property -> add value of "key" attribute or idShort to list of keys
131 key_prop = find_by_semantic_id(smc.value, "https://admin-shell.io/idta/AssetInterfacesDescription/1/0/key")
132 local_key = key_prop.value if key_prop else smc.id_short
133 new_key_path = key_path + [local_key]
134 else:
135 # TODO: use the key of the first-level property (or its idShort otherwise)
136 # is a top-level property
137 # key_prop: Property | None = find_by_semantic_id(
138 # smc.value, "https://admin-shell.io/idta/AssetInterfacesDescription/1/0/key"
139 # )
140 # local_key = key_prop.value if key_prop else smc.id_short
141 # new_key_path = [local_key]
142
143 new_key_path = key_path
144 # NOTE (Tom Gneuß, 2025-10-20)
145 # See GitHub Issue: https://github.com/admin-shell-io/submodel-templates/issues/197
146 # First-level properties are allowed to have a "key" attribute - otherwise the idShort path is used.
147 # However, complex first-level properties would represent, e.g., the JSON payload (object) itself.
148 # This JSON payload does only have keys for nested elements.
149 # So, using the key (or idShort) of the first-level property to get the JSON object from the payload
150 # is not possible.
151 # On the other hand: the first-level property could intentionally be something within the JSON object.
152 # In that case, having a "key" (or using the idSort) is entirely valid.
153 # How to distinguish both cases?
154
155 # create the idShort path of this property
156 full_path = f"{parent_path}.{smc.id_short}"
157 # add this property with all its details to the mapping -> href (from top-level parent if this is nested),
158 # protocol bindings (from top-level parent if this is nested), list of keys
159 mapping[full_path] = PropertyDetails(href, new_key_path, protocol_binding)
160
161 # traverse nested "properties" or "items"
162 for nested_sem_id in [
163 "https://www.w3.org/2019/wot/json-schema#properties",
164 "https://www.w3.org/2019/wot/json-schema#items",
165 ]:
166 nested_group: model.SubmodelElementCollection | None = find_by_semantic_id(smc.value, nested_sem_id)
167 if nested_group:
168 # attach the name of that SMC ("items" or "properties" or similar) to the key_path
169 full_path += "." + nested_group.id_short
170
171 # find all nested properties/items by semantic-ID
172 nested_properties: list[model.SubmodelElement] = find_all_by_semantic_id(
173 nested_group.value, "https://www.w3.org/2019/wot/json-schema#propertyName"
174 )
175
176 # traverse all nested properties/items recursively
177 for idx, nested in enumerate(nested_properties):
178 if nested_sem_id.endswith("#items"):
179 # for arrays: append index instead of "key" attribute
180 traverse_property(nested, full_path, href, new_key_path, is_items=True, idx=idx)
181 else:
182 traverse_property(nested, full_path, href, new_key_path)
183
184 # process all first-level properties
185 for fl_property in fl_properties: # type: SubmodelElementCollection
186 forms: model.SubmodelElementCollection | None = find_by_semantic_id(fl_property.value, "https://www.w3.org/2019/wot/td#hasForm")
187 if forms is None:
188 raise ValueError(f"'forms' SMC not found in '{fl_property.id_short}' SMC.")
189
190 href: model.Property | None = find_by_semantic_id(forms.value, "https://www.w3.org/2019/wot/hypermedia#hasTarget")
191 if href is None:
192 raise ValueError("'href' Property not found in 'forms' SMC.")
193
194 # get the href value of the first-level property
195 href_value = href.value
196
197 # construct the idShort path up to "Interface_.InteractionMetadata.properties"
198 # will be used as prefix for the idShort paths of the first-level and nested properties
199 idshort_path_prefix = f"{aid_interface.id_short}.{interaction_metadata.id_short}.{properties.id_short}"
200
201 # check which protocol-specific subtype of forms is used
202 # there is no clean solution for determining the subtype (e.g., a supplSemId)
203 # -> can only be figured out if the specific fields are present
204 protocol_binding: IProtocolBinding = None
205
206 # ... try HTTP ("htv_methodName" must be present)
207 htv_method_name: model.Property | None = find_by_semantic_id(forms.value, "https://www.w3.org/2011/http#methodName")
208 if htv_method_name is not None:
209 protocol_binding: HttpProtocolBinding = HttpProtocolBinding(htv_method_name.value, {})
210 htv_headers: model.SubmodelElementCollection | None = find_by_semantic_id(forms.value, "https://www.w3.org/2011/http#headers")
211 if htv_headers is not None:
212 for header in htv_headers.value: # type: ignore # type: SubmodelElementCollection
213 htv_field_name: model.Property | None = find_by_semantic_id(header.value, "https://www.w3.org/2011/http#fieldName")
214 htv_field_value: model.Property | None = find_by_semantic_id(header.value, "https://www.w3.org/2011/http#fieldValue")
215 protocol_binding.headers[htv_field_name.value] = htv_field_value.value
216
217 # TODO: the other protocols
218 # ... try Modbus
219 # ... try MQTT
220
221 # recursively parse the first-level property and its nested properties (if any)
222 traverse_property(
223 smc=fl_property,
224 parent_path=idshort_path_prefix,
225 href=href_value,
226 key_path=[],
227 is_items=False,
228 idx=None,
229 is_top_level=True,
230 protocol_binding=protocol_binding,
231 )
232
233 return mapping

235 def parse_security(self, aid_interface: model.SubmodelElementCollection) -> IAuthenticationDetails:
236 """Extract the authentication details (EndpointMetadata.security) from the provided interface in the AID.
237
238 :param aid_interface: An SMC describing an interface in the AID.
239 :return: A subtype of IAuthenticationDetails with details depending on the specified authentication method for the interface.
240 """
241 endpoint_metadata: model.SubmodelElementCollection | None = find_by_semantic_id(
242 aid_interface.value, "https://admin-shell.io/idta/AssetInterfacesDescription/1/0/EndpointMetadata"
243 )
244 if endpoint_metadata is None:
245 raise ValueError(f"'EndpointMetadata' SMC not found in the provided '{aid_interface.id_short}' SMC.")
246
247 security: model.SubmodelElementList | None = find_by_semantic_id(
248 endpoint_metadata.value, "https://www.w3.org/2019/wot/td#hasSecurityConfiguration"
249 )
250 if security is None:
251 raise ValueError("'security' SML not found in 'EndpointMetadata' SMC.")
252
253 # TODO: resolve the full reference(s)
254 # for now, assume there is only one reference to the security in use
255 # -> access SML[0]
256 # assume that this ReferenceElement points to a security scheme in this very AID SM
257 # -> can just use the last key to determine the type of security
258 sc_idshort = security.value[0].value.key[-1].value
259
260 # get the securityDefinitions SMC
261 security_definitions: model.SubmodelElementCollection | None = find_by_semantic_id(
262 endpoint_metadata.value, "https://www.w3.org/2019/wot/td#definesSecurityScheme"
263 )
264 if security_definitions is None:
265 raise ValueError("'securityDefinitions' SMC not found in 'EndpointMetadata' SMC.")
266
267 # find the security scheme SMC with the same idShort as mentioned in the reference "sc"
268 referenced_security: model.SubmodelElementCollection | None = find_by_id_short(security_definitions.value, sc_idshort)
269 if referenced_security is None:
270 raise ValueError(f"Referenced security scheme '{sc_idshort}' SMC not found in 'securityDefinitions' SMC")
271
272 # get the name of the security scheme
273 scheme: model.Property | None = find_by_semantic_id(referenced_security.value, "https://www.w3.org/2019/wot/security#SecurityScheme")
274 if scheme is None:
275 raise ValueError(f"'scheme' Property not found in referenced security scheme '{sc_idshort}' SMC.")
276
277 auth_details: IAuthenticationDetails = None
278
279 match scheme.value:
280 case "nosec":
281 auth_details = NoAuthenticationDetails()
282
283 case "basic":
284 basic_sc_name: model.Property | None = find_by_semantic_id(referenced_security.value, "https://www.w3.org/2019/wot/security#name")
285 if basic_sc_name is None:
286 raise ValueError("'name' Property not found in 'basic_sc' SMC")
287
288 auth_base64 = basic_sc_name.value
289 auth_plain = base64.b64decode(auth_base64).decode("utf-8")
290 auth_details = BasicAuthenticationDetails(auth_plain.split(":")[0], auth_plain.split(":")[1])
291
292 # TODO: remaining cases
293 case "digest":
294 pass
295 case "bearer":
296 pass
297 case "psk":
298 pass
299 case "oauth2":
300 pass
301 case "apikey":
302 pass
303 case "auto":
304 pass
305
306 return auth_details
