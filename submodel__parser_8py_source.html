10def get_submodel_element_by_id_short_path(submodel: model.Submodel, id_short_path: str) -> model.SubmodelElement:
11 """Retrieve a specific submodel element from the submodel at a specific idShort path.
12
13 :param submodel: The submodel to search within.
14 :param id_short_path: IdShort path to the submodel element (dot-separated), e.g., "Element1.Element2[0].Element3".
15 :return: The found submodel element or None if not found.
16 """
17 # Split the path by '.' and traverse the structure
18 parts = id_short_path.split(".")
19 current_elements = submodel.submodel_element
20 part_index = 0
21 for part in parts:
22 part_index += 1
23 # Handle indexed access like "Element[0]" for SubmodelElementLists
24 if "[" in part and "]" in part:
25 # Split SubmodelElementList name and index
26 base, idx = part[:-1].split("[")
27 idx = int(idx)
28 # Find the SubmodelElementList in the current elements
29 submodel_element = next((el for el in current_elements if el.id_short == base), None)
30
31 if not submodel_element or not (isinstance(submodel_element, (model.SubmodelElementList, model.SubmodelElementCollection))):
32 logger.debug(f"Submodel element '{base}' not found or is not a collection/list in current {current_elements}.")
33 return None
34
35 # Check if index is within range
36 if idx >= len(submodel_element.value):
37 logger.debug(f"Index '{idx}' out of range for element '{base}' with length {len(submodel_element.value)}.")
38 return None
39
40 # get the element by its index from SubmodelElementList
41 submodel_element = submodel_element.value[idx]
42
43 else:
44 # Find the SubmodelElement in the current SubmodelElementCollection
45 submodel_element = next((el for el in current_elements if el.id_short == part), None)
46
47 if not submodel_element:
48 logger.debug(f"Submodel element '{part}' not found in current {current_elements}.")
49 return None
50
51 # If we've reached the last part, return the found element
52 if part_index == len(parts):
53 return submodel_element
54
55 # If the found element is a collection or list, continue traversing
56 if isinstance(submodel_element, (model.SubmodelElementCollection, model.SubmodelElementList)):
57 current_elements = submodel_element.value
58 else:
