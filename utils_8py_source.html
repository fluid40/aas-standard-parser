35def _convert_to_object(content: dict) -> Any | None:
36 """Convert a dictionary to a BaSyx SDK framework object.
37
38 :param content: dictionary to convert
39 :return: BaSyx SDK framework object or None
40 """
41 if not content or len(content) == 0:
42 logger.debug("Empty content provided for conversion to object.")
43 return None
44
45 try:
46 dict_string = json.dumps(content)
47 return json.loads(dict_string, cls=basyx.aas.adapter.json.json_deserialization.AASFromJsonDecoder)
48 except Exception as e:
49 logger.error(f"Decoding error: {e}")
50 logger.error(f"In JSON: {content}")
51 return None

54def decode_base_64(text: str) -> str:
55 """Decode a Base64 encoded string.
56
57 :param text: Base64 encoded string to decode
58 :return: Decoded string
59 """
60 missing_padding = len(text) % 4
61 if missing_padding:
62 text += "=" * (4 - missing_padding)
